sf_bln_roads <- sf::read_sf(dsn = here::here("data", "osm_berlin_shp", "gis_osm_roads_free_1.shp"), 
                              layer = "gis_osm_roads_free_1")

sf_bln_water <- sf::read_sf(dsn = here::here("data", "osm_berlin_shp", "gis_osm_water_a_free_1.shp"), 
                            layer = "gis_osm_water_a_free_1")
sf_bln_roads_classified <- 
  sf_bln_roads %>% 
  filter(!fclass %in% c("footway", "steps")) %>% 
  filter(!is.na(name)) %>% 
  mutate(
    type = case_when(
      str_detect(name, "Strae|strae") ~ "Strae",
      str_detect(name, "Allee|allee") ~ "Allee",
      str_detect(name, "Platz|platz") ~ "Platz",
      str_detect(name, "Damm|damm") ~ "Damm",
      str_detect(name, "Chaussee|chaussee") ~ "Chaussee",
      str_detect(name, "Weg|weg") ~ "Weg",
      str_detect(name, "Ufer|ufer|Steg|steg|See|see") ~ "Water",  ## i.e. Ufer, Steg, See
      TRUE ~ "Other" ## e.g. Promenade, Pfad, Ring, Graben, Zeile, Gasse, Brcke, Hof, Bogen, Tor, ...
    ),
    type = factor(type, levels = c("Strae", "Weg", "Allee", "Damm", 
                                   "Platz", "Chaussee", "Water", "Other"))
  )
  
sf_bln_roads_classified %>% 
  ggplot() + 
    ## water
    geom_sf(data = sf_bln_water,
            fill = "#cedded",  
            color = "#cedded") +
    ## roads by type
    geom_sf(aes(color = type), 
            size = 0.3, 
            show.legend = "point") + 
     scale_color_manual(
       values = c(
         "#f6cf71",  ## Strae
         "#019868",  ## Weg
         "#ec0b88",  ## Allee
         "#651eac",  ## Damm
         "#e18a1e",  ## Platz
         "#9dd292",  ## Chaussee
         "#2b7de5",  ## Water
         "#c6c6c6"   ## Andere
       ), 
       name = NULL,
       labels = c("STRAE    \n(street)    ", "WEG    \n(path)    ", "ALLEE    \n(alley)    ", "DAMM    \n(dam)    ",
                  "PLATZ    \n(square)    ", "CHAUSSEE    \n(avenue)    ", "WATER-    \nRELATED    ", "OTHER\nNAMES")
       # labels = c("STRAE    ", "WEG    ", "ALLEE    ", "DAMM    ",

    ) +

         caption = ' Road type names that are related to close-by water bodies, i.e. **"See" (lake)**, **"Ufer" (shore)**, and **"Steg" (footbridge)**<br> Other road type names include, e.g. **"Promenade" (esplanade)**, **"Pfad" (trail)**, **"Ring" (ring road)**, **"Brcke" (bridge)**, **"Graben" (trench)**, and **"Hof" (courtyard)**') +
